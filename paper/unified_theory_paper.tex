\begin{abstract}
We present a formally verified framework that derives the Standard Model gauge group SU(3)$\times$SU(2)$\times$U(1), its complete one-generation fermion content, and all hypercharge ratios from seven explicit inputs: five structural primitives (source functional, Lie algebra, Lorentzian metric, linear composition, perturbation dimension $\geq 2$), one physical hypothesis (energy boundedness), and one selection principle (minimality of the fermion count). The derivation chain is: chirality (from the source/dressing decomposition) requires complex representations, which restricts the gauge algebra to SU($N$) type; the K/P structure forces two factors (one chiral, one vector-like) that must be non-isomorphic as represented algebras (proven: a surjective equivalence would transport the vector-like property, contradicting chirality); the representation structure is derived from color parity and chirality (the only chiral color-parity-satisfying structure is $(N_c, N_w) + N_w \times (\bar{N}_c, 1) + (1, N_w) + (1,1)$); charge determinacy uniquely selects SU(2) as the weak factor ($N_w = 2$ is the only value giving fully determined hypercharges); distinctness and minimality select SU(3) as the color factor; U(1) arises independently from the dressing sector; and anomaly cancellation --- via a universal cubic factorization $6N_c(N_c - r - 1)(N_c + r + 1) = 0$ valid for any SU($N_c$) --- uniquely determines all hypercharge ratios as $(1, -4, 2, -3, 6) \cdot y_Q$.

The framework additionally derives: the Einstein field equation as the unique second-order gravitational dynamics in 4D (with the second-order condition from stability via the Ostrogradsky instability theorem); full nonabelian Yang--Mills structure; the propagation rule $e^{ikL}$ from the source functional's linearity (yielding an unbroken chain from primitives to interference); the Born-type observable; the Higgs potential as the unique quartic rotation-invariant on the K/P space; symmetry breaking from decoherence; the graviton as a traceless P-sector defect with 2 polarizations in $d = 3$; minimal coupling; a strong CP resolution; and $N_g \geq 3$ from CP violation. A zero-parameter prediction gives $\alpha_2(M_Z) \approx 0.031$ (measured: 0.034, 9\% error). The algebraic core is formally verified in Lean~4 with zero custom axioms and zero \texttt{sorry} statements.
\end{abstract}

\begin{theorem}[Chirality from K/P]

\begin{theorem}[Complex representations required]
\label{thm:complex_reps}
Chirality requires left-handed and right-handed fermions to transform in inequivalent representations. This is only possible if the gauge algebra has \emph{complex} representations (representations not isomorphic to their conjugates). Among compact simple Lie algebras, the classification of representation types is:

Minimality of the fundamental representation dimension selects SU($N$): $\dim = N$ for SU($N$), versus $\dim = 16$ for SO(10) and $\dim = 27$ for E$_6$. The chiral gauge algebra factors are therefore of SU($N$) type. (For the classification of representation types, see Fulton--Harris~\cite{fultonharris}.)
\end{theorem}

\begin{theorem}[Two factors from K/P structure]
\label{thm:two_factors}
The K/P decomposition requires two kinds of gauge action: chiral (different on K and P) and vector-like (same on both). A single simple factor provides only one kind. Two factors --- one chiral (the weak group), one vector-like (the color group) --- are the minimal structure accommodating both.
\end{theorem}

\begin{theorem}[Representation structure forced]
\label{thm:rep_structure}
The fermion representation content $(N_c, N_w) + N_w \times (\bar{N}_c, 1) + (1, N_w) + (1, 1)$ is \emph{derived}, not assumed. Color parity (vanishing of the SU$(N_c)^3$ anomaly) requires equal positive and negative contributions. If both colored sectors carry SU$(N_w)$ multiplets, color parity forces equal multiplicities, making the theory vector-like (contradicting chirality). The unique chiral solution has one $(N_c, N_w)$ multiplet and $N_w$ copies of $(\bar{N}_c, 1)$, with the lepton sector $(1, N_w) + (1, 1)$ required for the mixed anomaly.
\end{theorem}

\begin{theorem}[Distinctness from chirality]
\label{thm:distinctness}
The weak factor acts \emph{chirally} on $V = K \oplus P$: its restriction to $K$ is inequivalent to its restriction to $P$ (from Theorem~\ref{thm:chirality}). The color factor acts \emph{vector-like}: gauge invariance of the source functional $\varphi$ forces the color action to commute with the K/P projection, so its restrictions to $K$ and $P$ are equivalent. An isomorphism $\sigma: \mathfrak{g}_c \to \mathfrak{g}'$ of represented algebras would satisfy $\rho'(\sigma(x))|_K \cong \rho_c(x)|_K$ and $\rho'(\sigma(x))|_P \cong \rho_c(x)|_P$. Since $\rho_c(x)|_K \cong \rho_c(x)|_P$ (vector-like) and $\sigma$ is surjective, this yields $\rho'(y)|_K \cong \rho'(y)|_P$ for all $y \in \mathfrak{g}'$, contradicting chirality. Therefore $G_c \not\cong G'$.
\end{theorem}

\begin{theorem}[U(1) from dressing]
\label{thm:u1}
The complex structure $z = Q + iP$ forced by Frobenius carries a phase rotation $z \mapsto e^{i\theta}z$: a U(1) symmetry with a different geometric origin than the nonabelian factors.
\end{theorem}

\begin{theorem}[SU(2) from charge determinacy and minimality]
\label{thm:su2}
The fermion count for a weak factor $G' = \mathrm{SU}(N_w)$ is $7N_w + 1$ (from color parity + lepton sector counting). For $N_w = 1$ the theory is vector-like (excluded by chirality). Among chiral values $N_w \geq 2$, the charge variable count is $N_w + 3$ while the anomaly conditions number 4. $N_w = 2$ is the \emph{unique} value where the free parameters equal 1 (overall normalization), making all hypercharge ratios fully determined. For $N_w \geq 3$ there are $\geq 2$ free parameters (underdetermined). Thus $N_w = 2$ is uniquely selected, giving 15 fermions.
\end{theorem}

\begin{theorem}[SU(3) from distinctness and minimality]
\label{thm:su3}
The fermion count for color group SU($N_c$) is $4N_c + 3$. The minimum is at $N_c = 2$ (11 fermions), but $N_c = 2$ gives $G_c = \mathrm{SU}(2) \cong G'$, violating distinctness (Theorem~\ref{thm:distinctness}). The next minimum at $N_c = 3$ gives 15 fermions and $G_c = \mathrm{SU}(3) \not\cong \mathrm{SU}(2)$. SU(3) is uniquely selected.
\end{theorem}

\begin{theorem}[Universal cubic factorization]
\label{thm:universal_cubic}
For any SU($N_c$) $\times$ SU(2) $\times$ U(1), the cubic anomaly condition factors as:
\[
6\,N_c\,(N_c - r - 1)(N_c + r + 1) = 0,

giving hypercharges $Y_Q \cdot (1, -(N_c+1), N_c - 1, -N_c, 2N_c)$. At $N_c = 3$: $(1, -4, 2, -3, 6) \cdot y_Q$ --- exactly the Standard Model. The factorization is universal; the SM charges are the unique solution of a structural equation, not an accident of 3 colors.
\end{theorem}
